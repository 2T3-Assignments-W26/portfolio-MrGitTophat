```C#
        static Int32 Selection = 0;
        // static bool Loop = true;

        {
            do
            {
                switch (Selection)
                {
                    case 0:
                        Menu();
                        break;
                    case 1:
                        Rectangle();
                        break;
                    case 2:
                        Collatz();
                        break;
                    case 3:
                        Dollar();
                        break;
                    case 4:
                        Oldest();
                        break;
                    case 5:
                        HiLo();
                        break;
                    case 6:

                        break;
                }
            } while (Selection !< 6); // Any number that is 6 or above will end the program
            Selection = 0;
            Console.WriteLine("Thank you for using this program");
        }
```
